122.Does the total amount in orders by customer in Rome and London exceed the commission paid to salespersons in London and New York by more than 5 times?

select (select sum(amt*c.comm) as totalsales from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum and (c.city='London' or c.city='New York'))as tot_order,

(select sum(a.amt) from orders a,customer b wherea.cnum=b.cnum and (b.city='Rome' or b.city='London')) as tot_comm,

(select sum(a.amt) from orders a,customer b wherea.cnum=b.cnum and (b.city='Rome' or b.city='London'))/select (select sum(amt*c.comm) as totalsales from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum and (c.city='London' or c.city='New York'))as division;

123.Which is the date, order number, amt and city for each salesperson (by name) for the maximum order he has obtained?
Select a.odate,a.onum,a.amt,b.city from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum and a.amt in(select max(a.amt) from orders a,salespeople b,customer c where a.cnum=c.cnum and b.snum=c.snum group by b.sname);

124.Which salesperson(s) should be fired?
select b.snum,c.sname,sum(a.amt) as totalsalesbySP from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum group by b.snum order by totalsalesbySP limit 1;

126.What is the gross profit for the company?
select (select sum(amt) from orders),(select sum(c.comm*a.amt) as 'Total Commission' from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum) as gross_Profit;

127. Find the contribution of each salesperson in percentage.
select b.snum,c.sname,sum(a.amt) as totalsalesbySP sum(amt)*c.comm,sum((a.amt*100)/(select sum(amt) from orders) as SalesPercent from orders a,customer b,salespeople c where a.cnum=b.cnum and b.snum=c.snum group by b.snum order by totalsalesbySP desc;
